python - <<'EOF'
import json

def walk_vss(node, prefix=""):
    out = []
    for name, child in node.get("children", {}).items():
        p = f"{prefix}.{name}" if prefix else name
        if "datatype" in child:
            out.append(p)
        out.extend(walk_vss(child, p))
    return out

tree = json.load(open("vss.json"))
paths = walk_vss(tree["Vehicle"], "Vehicle")

print("First 20 signals:")
for p in paths[:20]:
    print(" ", p)

print("\nTotal signals:", len(paths))
EOF
